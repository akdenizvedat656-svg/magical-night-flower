using UnityEngine; using UnityEngine.SceneManagement; public enum GameState { Quiz, Shooting, GameOver } public class GameManager : MonoBehaviour { public static GameManager Instance; public GameState CurrentState = GameState.Quiz; [SerializeField] GameObject quizCanvas; [SerializeField] GameObject shooterCanvas; // veya player controller enable/disable [SerializeField] QuestionSO[] questions; private int currentQuestionIndex = 0; public int score = 0; public int lives = 3; void Awake() => Instance = this; void Start() { ShowQuiz(); } public void ShowQuiz() { CurrentState = GameState.Quiz; quizCanvas.SetActive(true); // shooterCanvas.SetActive(false); veya PlayerController.enabled = false; DisplayQuestion(); } void DisplayQuestion() { if (currentQuestionIndex >= questions.Length) { // Tüm sorular bitti → GameOver veya loop CurrentState = GameState.GameOver; return; } var q = questions[currentQuestionIndex]; // UI'ya soru ve şıkları yaz (TextMeshPro vs.) // QuizU sample'dan UI Toolkit alabilirsin: https://unity.com/demos (QuizU) } public void OnAnswerSelected(int selectedIndex) { var correct = questions[currentQuestionIndex].correctAnswerIndex; if (selectedIndex == correct) { score += questions[currentQuestionIndex].points; // 500/1000/5000 SpawnEnemy(questions[currentQuestionIndex].points); CurrentState = GameState.Shooting; quizCanvas.SetActive(false); // shooterCanvas.SetActive(true); PlayerController.enabled = true; } else { lives--; if (lives <= 0) CurrentState = GameState.GameOver; else currentQuestionIndex++; // veya tekrar soru } } void SpawnEnemy(int value) { // Shooter reposundaki enemy spawn fonksiyonunu çağır // HP = value / 100 gibi ayarla // Öldüğünde → currentQuestionIndex++; ShowQuiz(); } // Update'de state'e göre input kontrolü vs. }
